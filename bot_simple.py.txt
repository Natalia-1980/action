def init_db():
    conn = sqlite3.connect('users.db')
    c = conn.cursor()
    c.execute('''CREATE TABLE IF NOT EXISTS users
                 (user_id INTEGER PRIMARY KEY,
                  current_day INTEGER DEFAULT 1)''')
    c.execute('''CREATE TABLE IF NOT EXISTS answers
                 (user_id INTEGER,
                  day_number INTEGER,
                  is_correct INTEGER,
                  PRIMARY KEY (user_id, day_number))''')
    conn.commit()
    conn.close()

def get_user(user_id):
    conn = sqlite3.connect('users.db')
    c = conn.cursor()
    c.execute("SELECT current_day FROM users WHERE user_id = ?", (user_id,))
    result = c.fetchone()
    conn.close()
    return result

def create_user(user_id):
    conn = sqlite3.connect('users.db')
    c = conn.cursor()
    c.execute("INSERT INTO users (user_id, current_day) VALUES (?, 1)", (user_id,))
    conn.commit()
    conn.close()

def set_user_day(user_id, day):
    conn = sqlite3.connect('users.db')
    c = conn.cursor()
    c.execute("UPDATE users SET current_day = ? WHERE user_id = ?", (day, user_id))
    conn.commit()
    conn.close()

def save_answer(user_id, day, is_correct):
    conn = sqlite3.connect('users.db')
    c = conn.cursor()
    c.execute("INSERT OR REPLACE INTO answers (user_id, day_number, is_correct) VALUES (?, ?, ?)",
              (user_id, day, 1 if is_correct else 0))
    conn.commit()
    conn.close()

def get_answer_status(user_id, day):
    conn = sqlite3.connect('users.db')
    c = conn.cursor()
    c.execute("SELECT is_correct FROM answers WHERE user_id = ? AND day_number = ?", (user_id, day))
    result = c.fetchone()
    conn.close()
    return result

def get_stats(user_id):
    conn = sqlite3.connect('users.db')
    c = conn.cursor()
    c.execute("SELECT COUNT(*), SUM(is_correct) FROM answers WHERE user_id = ?", (user_id,))
    total, correct = c.fetchone()
    conn.close()
    return (total or 0, correct or 0)

def get_main_keyboard():
    buttons = [[KeyboardButton(text="📅 /day"), KeyboardButton(text="📊 /stats")]]
    return ReplyKeyboardMarkup(keyboard=buttons, resize_keyboard=True)

def get_answer_keyboard():
    buttons = [[KeyboardButton(text="1"), KeyboardButton(text="2"), KeyboardButton(text="3")]]
    return ReplyKeyboardMarkup(keyboard=buttons, resize_keyboard=True)

# ========== КОМАНДЫ ==========
bot = Bot(token=API_TOKEN)
dp = Dispatcher()

@dp.message(Command("start"))
async def cmd_start(message: types.Message):
    user_id = message.from_user.id
    if not get_user(user_id):
        create_user(user_id)
    await message.answer("👋 Привет! Я бот по ЭТрН.\n\nОтправь /day чтобы получить ситуацию.", reply_markup=get_main_keyboard())

@dp.message(Command("day"))
async def cmd_day(message: types.Message):
    user_id = message.from_user.id
    user = get_user(user_id)
    if not user:
        create_user(user_id)
        user = (1,)
    
    current_day = user[0]
    if current_day > 30:
        await message.answer("🎉 Вы прошли все 30 дней!")
        return
    
    data = SITUATIONS[current_day]

@dp.message(Command("stats"))
async def cmd_stats(message: types.Message):
    user_id = message.from_user.id
    total, correct = get_stats(user_id)
    if total > 0:
        percent = int(correct / total * 100)
        await message.answer(f"📊 Статистика:\n✅ Правильно: {correct}/{total}\n📈 Процент: {percent}%")
    else:
        await message.answer("📊 У вас пока нет ответов. Отправьте /day")

@dp.message()
async def handle_answer(message: types.Message):
    user_id = message.from_user.id
    user = get_user(user_id)
    if not user:
        await cmd_start(message)
        return
    
    current_day = user[0]
    if current_day > 30:
        return
    
    if message.text not in ["1", "2", "3"]:
        await message.answer("Отправьте цифру 1, 2 или 3")
        return
    
    # Проверяем, не отвечал ли уже
    if get_answer_status(user_id, current_day):
        await message.answer(f"Вы уже ответили на день {current_day}. Используйте /day для следующего дня.")
        return
    
    choice = int(message.text)
    data = SITUATIONS[current_day]
    is_correct = (choice == data['correct'])
    
    save_answer(user_id, current_day, is_correct)
    
    if is_correct:
        await message.answer(f"✅ Правильно!\n\n📖 {data['explanation']}\n\n⚖️ {data['norm']}")
    else:
        correct_num = {1: "первый", 2: "второй", 3: "третий"}[data['correct']]
        await message.answer(f"❌ Неправильно. Правильный ответ - {correct_num}.\n\n📖 {data['explanation']}\n\n⚖️ {data['norm']}")
    
    # Переходим к следующему дню
    next_day = current_day + 1
    if next_day <= 30:
        set_user_day(user_id, next_day)
        await message.answer(f"📅 Теперь вы на дне {next_day}. Отправьте /day чтобы продолжить.")
    else:
        await message.answer("🎉 Поздравляю! Вы прошли все 30 дней!")
